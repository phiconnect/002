\begin{question}
  What is the value of $\sqrt{144}$?
  \begin{choices}
    \choice $10$
    \choice $11$
    \choice $12$
    \choice $13$
  \end{choices}
  \answer{$12$}
\end{question}

\begin{question}
  Simplify: $\dfrac{3x^2 - 12}{3}$
  \begin{choices}
    \choice $x^2 - 4$
    \choice $x^2 - 12$
    \choice $3x^2 - 4$
    \choice $x - 4$
  \end{choices}
  \answer{$x^2 - 4$}
\end{question}

\begin{problem}
  Solve for $x$: $2x + 5 = 17$
  \answer{6}
\end{problem}

\begin{exercise}
  What is the derivative of $f(x) = x^3 - 4x + 2$?
  \begin{choices}
    \choice $3x^2 - 4$
    \choice $3x^2 + 4$
    \choice $x^2 - 4$
    \choice $3x - 4$
  \end{choices}
  \answer{$3x^2 - 4$}
\end{exercise}

\begin{question}
  Evaluate $\displaystyle\int_0^1 2x\,dx$.
  \answer{1}
\end{question}

\begin{question}
  If $\sin\theta = \frac{1}{2}$, what is $\theta$ in degrees (for $0 \le \theta \le 90$)?
  \begin{choices}
    \choice $30°$
    \choice $45°$
    \choice $60°$
    \choice $90°$
  \end{choices}
  \answer{$30°$}
\end{question}
